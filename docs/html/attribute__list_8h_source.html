165template <class AttrName, class... AttrPtrs, class V>
166constexpr auto take_attr(auto& t, const std::tuple<AttrPtrs...>& ptrs, auto&& set_attrs, V&& value) {
167 if constexpr (AttrName::value) {
168 static_assert(attributes_contain_v<AttrName::name, AttrPtrs...>,
169 "attribute does not apply to the specified element");
170 constexpr auto attr_holder = find_attr_holder<AttrName::name, AttrPtrs...>::interface_name;
171
172 using attr_ptr_type = std::remove_reference_t<V> attribute_list<attr_holder>::*;
173 t.*(std::get<attr_ptr_type>(ptrs)) = value;
174 return std::tuple_cat(std::move(set_attrs), std::make_tuple(std::get<attr_ptr_type>(ptrs)));
175 } else {
176 return set_attrs;
177 }
178};

203template <static_string In, class... AttrPtrs, class V, class... R, class... SetAttrs, class... Children>
204constexpr auto initialize_aspect(attribute_list<In>& t, const std::tuple<AttrPtrs...>& ptrs,
205 std::tuple<SetAttrs...>&& set_attrs, std::tuple<Children...>&& children, V&& value, R&&... r) {
206 using attr_name = get_attribute_name<std::remove_cvref_t<V>>;

208 if constexpr (sizeof...(R) == 0) {
209 return element_aspects{std::move(take_attr<attr_name>(t, ptrs, set_attrs, std::forward<V>(value))),
210 std::move(take_child<attr_name>(t, std::move(children), std::forward<V>(value)))};
211 } else {
212 return initialize_aspect(t, ptrs,

214 std::move(take_child<attr_name>(t, std::move(children), std::forward<V>(value))),
